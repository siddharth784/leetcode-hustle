class Solution {
    public int countGoodSubstrings(String s) {
        if(s.length()<3)
            return 0;
        if(s.length()==3)
            return unique(s);
        return (countGoodSubstrings(s.substring(1))+unique(s.substring(0,3)));
    }
    
    int unique(String s){
        if(s.charAt(0)!=s.charAt(1) && s.charAt(1)!=s.charAt(2) && s.charAt(0)!=s.charAt(2))
            return 1;
        else
            return 0;
    }
}
