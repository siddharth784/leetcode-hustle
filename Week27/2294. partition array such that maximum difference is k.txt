class Solution {
    public int partitionArray(int[] nums, int k) {
        Arrays.sort(nums);
        int count=0;
        int i=0,j=0;
        while(i<nums.length && j<nums.length){
            j=i;
            while(j<nums.length && nums[j]-nums[i]<=k)
                j++;
                j--;
            count++;
            i=j+1;
        }
        return count;

            
    }
}
