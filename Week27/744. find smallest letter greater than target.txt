class Solution {
    public char nextGreatestLetter(char[] letters, char target) {
        if(letters[letters.length-1]<=target)
            return letters[0];
        int n=letters.length;
        int lo=0,hi=n-1;
        char ans='z';
        while(lo<=hi){
            int mid=(lo+hi)/2;
            if(target<letters[mid]){
                if(ans>letters[mid])
                    ans=letters[mid];
                hi=mid-1;
            }
            else
lo=mid+1;
    }
        return ans;
    }
}
