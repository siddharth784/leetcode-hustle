class Solution {
    public String removeStars(String s) {
        Stack<Character> st=new Stack<>();
        for(char ch:s.toCharArray()){
            if(ch!='*')
                st.push(ch);
            else
                st.pop();            
        }
        String ans="";
        while(!st.empty())
            ans=st.pop()+ans;
        return ans;
    }
}
