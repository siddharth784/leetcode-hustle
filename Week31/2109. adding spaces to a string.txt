class Solution {
    public String addSpaces(String s, int[] spaces) {
        int j=0;
        StringBuilder ans=new StringBuilder();
        if(spaces[0]==0){
            ans.append(" ");
            j++;
        }
        for(int i=0;i<s.length();i++){
            ans.append(s.charAt(i));
            if(j<spaces.length && spaces[j]-1==i){
                ans.append(" ");
                j++;
            }
        }
        return ans.toString();
    }
}
