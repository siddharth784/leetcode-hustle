class Solution {
    public int maxSum(int[][] grid) {
        int m=grid.length,n=grid[0].length,maxsum=0;
        for(int i=0;i<=m-3;i++){
            for(int j=0;j<=n-3;j++)
                maxsum=Math.max(maxsum,grid[i][j]+grid[i][j+1]+grid[i][j+2]+grid[i+1][j+1]+grid[i+2][j]+grid[i+2][j+1]+grid[i+2][j+2]);
        }
        
        return maxsum;
    }
}
