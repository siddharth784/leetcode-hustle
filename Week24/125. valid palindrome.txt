class Solution {
    public boolean isPalindrome(String s) {
        s=s.toLowerCase();
        char[] str=s.toCharArray();
        int i=0,j=str.length-1;
        while(i<j){
            if(!Character.isLetter(str[i]) && !Character.isDigit(str[i]))
                i++;
            else if(!Character.isLetter(str[j]) && !Character.isDigit(str[j]))
                j--;
            else if(str[i]!=str[j])
                return false;
            else if(str[i]==str[j]){
                i++;
                j--;
            }
        }
        return true;
    }
}
