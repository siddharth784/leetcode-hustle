class Solution {
    public int[][] restoreMatrix(int[] rowSum, int[] colSum) {
        //GREEDY+OBSERVE TESTCASES
        int m=rowSum.length,n=colSum.length;
        int[][] mat=new int[m][n];
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                int x=Math.min(rowSum[i],colSum[j]);
                mat[i][j]=x;
                rowSum[i]-=x;
                colSum[j]-=x;
            }
        }
        return mat;
        
    }
}
