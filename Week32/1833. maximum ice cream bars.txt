class Solution {
    public int maxIceCream(int[] costs, int coins) {
        Arrays.sort(costs);
        int count=0,spent=0;
        for(int i=0;i<costs.length;i++){
            if(spent+costs[i]<=coins){
                spent+=costs[i];
                count++;
            }
            else
                break;
        }
        return count;
    }
}
