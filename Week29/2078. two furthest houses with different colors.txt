class Solution {
    public int maxDistance(int[] colors) {
        int maxdis=0;
        for(int i=0;i<colors.length;i++){
            for(int j=colors.length-1;j>i && j-i>maxdis;j--){
                if(colors[i]!=colors[j]){
                    maxdis=Math.max(maxdis,j-i);
                    break;
                }
            }
        }
        
        return maxdis;
    }
}
