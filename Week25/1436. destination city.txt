class Solution {
    public String destCity(List<List<String>> paths) {
        HashMap<String,String> hm=new HashMap<>();
        for(List<String> path:paths){
            hm.put(path.get(0),path.get(1));
        }
        for(List<String> path:paths){
            if(!hm.containsKey(path.get(0)))
                return path.get(0);
            if(!hm.containsKey(path.get(1)))
                return path.get(1);
        }
        return "";
    }
}
