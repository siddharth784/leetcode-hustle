class Solution {
    public int maxAscendingSum(int[] nums) {
        int maxsum=nums[0];
        int currsum=nums[0];
        for(int i=1;i<nums.length;i++){
            if(nums[i]>nums[i-1]){
                currsum+=nums[i];
                maxsum=Math.max(currsum,maxsum);
            }
            else
                currsum=nums[i];
        }
        return maxsum;
    }
}
