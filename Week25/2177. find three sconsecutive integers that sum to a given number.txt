class Solution {
    public long[] sumOfThree(long num) {
        if(num%3!=0)
            return new long[0];
        long[] ans=new long[3];
        
        // k,k+1,k+2......3k+3
        
        num/=3;
        ans[0]=num-1;
        ans[1]=num;
        ans[2]=num+1;
        return ans;
    }
}
